class Solution {
public:
    int getDecimalValue(ListNode* head) {
        
        vector <long long> v;
        
        while(head != NULL)
        {
            v.push_back(head->val);
            head = head->next;
        }
        
        reverse(v.begin(), v.end());
        
        long long dec = 0;
        for(int i = 0; i < v.size(); i++)
        {
            dec += v[i] * pow(2, i);
        }
        
        return dec;
    }
};
